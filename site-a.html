-- Vertical Fragmentation of Tourists
CREATE TABLE Tourist_Basic AS
  SELECT TouristID, Name FROM Tourists;
 
CREATE TABLE Tourist_Contact AS
  SELECT TouristID, Nationality, Contact FROM Tourists;
 
-- Horizontal Fragmentation of Trips
CREATE TABLE Trips_North AS
  SELECT * FROM Trips WHERE Region = N'الشمال';
 
-- Mixed Fragmentation of Bookings
CREATE TABLE Booking_Info AS
  SELECT BookingID, TouristID, TripID FROM Bookings;
 
CREATE TABLE Booking_Amount AS
  SELECT BookingID, Amount FROM Bookings;
 
-- DB Links to other sites
CREATE DATABASE LINK south_db
  CONNECT TO south_user IDENTIFIED BY south_pass
  USING '(DESCRIPTION=...FREEPDB1...)';
 
-- Global View
CREATE VIEW All_Trips AS
  SELECT * FROM Trips
  UNION
  SELECT * FROM Trips@south_db
  UNION
  SELECT * FROM Trips@east_db;
